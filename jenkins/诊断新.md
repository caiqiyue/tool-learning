set ip "121.196.229.207"
set pass "Strong@365"
set timeout 300

set src "$env(WORKSPACE)/sales-diagnosis-system/"
set dest "/root/lixiang/new-diagnostic-system/sales-diagnosis-system/"

if {![file isdirectory $src]} {
    puts "ERROR: source directory not found: $src"
    exit 2
}

set rsync_cmd [list rsync -azvP --delete \
    --exclude ".git/" \
    --exclude "logs/" \
    --exclude "__pycache__/" \
    --exclude "*.pyc" \
    --exclude "backend/logs/" \
    --exclude "backend/uploads/" \
    --exclude "frontend/node_modules/" \
    --exclude "frontend/build/" \
    -e "ssh -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null" \
    $src "root@$ip:$dest" \
]

eval spawn $rsync_cmd

expect {
    -re "(yes/no)"      {send "yes\r"; exp_continue}
    -re "(?i)password:" {send "$pass\r"; exp_continue}
    timeout             {puts "ERROR: rsync timeout"; exit 255}
    eof
}

catch wait result
set exit_status [lindex $result 3]
exit $exit_status

#!/usr/bin/expect -f
set ip "121.196.229.207"
set pass "Strong@365"
set timeout 1200

set project_dir "/root/lixiang/new-diagnostic-system/sales-diagnosis-system"
set backend_dir  "$project_dir/backend"
set frontend_dir "$project_dir/frontend"

set envbin "/root/miniconda3/envs/sales-diagnosis/bin"
set py     "$envbin/python"
set npm    "$envbin/npm"
set node   "$envbin/node"

set backend_service "sales-diagnosis-backend"

set remote_cmd [format {bash -lc '
set -e

export PATH="%s:$PATH"

echo "=== Stop backend service ==="
systemctl stop %s || true

echo "=== Backend: install deps ==="
cd "%s"
%s -m pip install -U pip
if test -f requirements.txt; then
  %s -m pip install -r requirements.txt
else
  echo "No requirements.txt, skip"
fi

echo "=== Frontend: clean & install deps ==="
cd "%s"
%s -v

rm -rf node_modules/.cache || true
rm -rf node_modules || true

%s config set fund false
%s config set audit false
if test -f package-lock.json; then
  %s ci
else
  %s install
fi

echo "=== Frontend: build production ==="
%s run build

if [ ! -f "build/index.html" ]; then
  echo "ERROR: build/index.html not found!"
  exit 1
fi

echo "=== Reload systemd and restart backend ==="
systemctl daemon-reload
systemctl restart %s

echo "=== Optional web server reload ==="
if systemctl list-unit-files | grep -q "^nginx.service"; then
  systemctl reload nginx || systemctl restart nginx
elif systemctl list-unit-files | grep -q "^openresty.service"; then
  systemctl reload openresty || systemctl restart openresty
else
  echo "INFO: no nginx/openresty systemd unit found, skip web server reload"
fi

echo "=== Status ==="
systemctl status %s --no-pager || true

echo "=== Deployment completed successfully ==="
'} \
$envbin \
$backend_service \
$backend_dir $py $py \
$frontend_dir $node \
$npm $npm $npm $npm \
$npm \
$backend_service \
$backend_service]

spawn ssh -tt -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null root@$ip $remote_cmd

expect {
    -re "(yes/no)"           {send "yes\r"; exp_continue}
    -re "(?i)password:"      {send "$pass\r"; exp_continue}
    -re "Permission denied"  {puts "ERROR: Permission denied"; exit 255}
    timeout                  {puts "ERROR: ssh timeout"; exit 255}
    eof
}

catch wait result
set exit_status [lindex $result 3]
exit $exit_status
