#!/usr/bin/env bash
set -euo pipefail

die() { echo "ERROR: $*" >&2; exit 2; }

# ====== 必填：Jenkins 参数 ======
: "${RELEASE_TAG:?ERROR: RELEASE_TAG is required (e.g. v0.0.7)}"
TAG="${RELEASE_TAG}"

# ====== 固定与你当前仓库一致：origin/master ======
PRIMARY_REMOTE="${PRIMARY_REMOTE:-origin}"
PRIMARY_BRANCH="${PRIMARY_BRANCH:-master}"

# ====== Tag 格式（默认只允许 vX.Y.Z）======
TAG_REGEX="${TAG_REGEX:-^v[0-9]+\.[0-9]+\.[0-9]+$}"

echo "INFO: RELEASE_TAG=${TAG}"
echo "INFO: PRIMARY=${PRIMARY_REMOTE}/${PRIMARY_BRANCH}"

# （可选）输出 remote，方便排查 Jenkins 是否真的创建了 origin
git remote -v || true

# 1) tag 格式校验
if [[ ! "$TAG" =~ $TAG_REGEX ]]; then
  die "invalid RELEASE_TAG format: '${TAG}' (expected regex: ${TAG_REGEX})"
fi

# 2) tag 必须存在且指向 commit
git rev-parse -q --verify "refs/tags/${TAG}^{commit}" >/dev/null \
  || die "tag not found or not a commit: ${TAG}"
TAG_COMMIT="$(git rev-parse "refs/tags/${TAG}^{commit}")"

# 3) 不再 fetch：直接使用 Jenkins SCM 已经拉下来的远端主分支引用
MASTER_REF="refs/remotes/${PRIMARY_REMOTE}/${PRIMARY_BRANCH}"
git rev-parse -q --verify "${MASTER_REF}^{commit}" >/dev/null \
  || die "primary ref not found locally: ${MASTER_REF}. HINT: set SCM Refspec to fetch it (e.g. +refs/heads/master:${MASTER_REF})."

PRIMARY_COMMIT="$(git rev-parse "${MASTER_REF}^{commit}")"

# 4) 校验：tag commit 必须在 master 历史上
if git merge-base --is-ancestor "${TAG_COMMIT}" "${PRIMARY_COMMIT}"; then
  echo "OK: ${TAG} (${TAG_COMMIT}) is contained in ${MASTER_REF} (${PRIMARY_COMMIT})"
else
  die "${TAG} (${TAG_COMMIT}) is NOT contained in ${MASTER_REF} (${PRIMARY_COMMIT}). Refusing to deploy."
fi

# 5) 输出当前 HEAD 信息，帮助确认 Jenkins 真的检出了 tag
echo "INFO: Current HEAD: $(git rev-parse --short=12 HEAD)"
echo "INFO: Current branch: $(git rev-parse --abbrev-ref HEAD 2>/dev/null || true)"

#!/usr/bin/expect -f
set ip "121.196.229.207"
set pass "Strong@365"
set timeout 180

# Jenkins 工作空间是 recommend-system 仓库根目录
set src "$env(WORKSPACE)/"
set dest "/root/caiqiyue_files/recommend-system/"

if {![file isdirectory $src]} {
    puts "ERROR: source directory not found: $src"
    exit 2
}

set rsync_cmd [list rsync -azvP --delete \
    --exclude ".git/" \
    --exclude ".idea/" \
    --exclude ".vscode/" \
    --exclude "__pycache__/" \
    --exclude "*.pyc" \
    --exclude "logs/" \
    --exclude "media/" \
    --exclude "recommend_system_project/media/" \
    --exclude "recommend_system_project/staticfiles/" \
    -e "ssh -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null" \
    $src "root@$ip:$dest" \
]

eval spawn $rsync_cmd

expect {
    -re "(yes/no)"      {send "yes\r"; exp_continue}
    -re "(?i)password:" {send "$pass\r"; exp_continue}
    timeout             {puts "ERROR: rsync timeout"; exit 255}
    eof
}

catch wait result
set exit_status [lindex $result 3]
exit $exit_status

#!/usr/bin/expect -f
set ip "121.196.229.207"
set pass "Strong@365"
set timeout 600

set remote_root "/root/caiqiyue_files/recommend-system"
set app_dir "/root/caiqiyue_files/recommend-system/recommend_system_project"
set env_file "/root/caiqiyue_files/recommend-system/app.env"
set conda_env "recommend_prj"
set service_name "recommend_system"

set remote_cmd "bash -lc ' \
set -e; \
test -d $app_dir || { echo APP_DIR_NOT_FOUND: $app_dir; exit 2; }; \
test -f $env_file || { echo ENV_FILE_NOT_FOUND: $env_file; exit 2; }; \
cd $app_dir; \
source /root/miniconda3/etc/profile.d/conda.sh; \
conda activate $conda_env; \
export RECOMMEND_SYSTEM_ENV_FILE=$env_file; \
python -m pip install -U pip; \
python -m pip install -r requirements.txt; \
python manage.py check; \
python manage.py collectstatic --noinput; \
systemctl daemon-reload; \
systemctl restart $service_name; \
systemctl is-active $service_name; \
systemctl status $service_name --no-pager \
'"

spawn ssh -tt -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null root@$ip $remote_cmd

expect {
    -re "(yes/no)"           {send "yes\r"; exp_continue}
    -re "(?i)password:"      {send "$pass\r"; exp_continue}
    -re "Permission denied"  {puts "ERROR: Permission denied"; exit 255}
    timeout                  {puts "ERROR: ssh timeout"; exit 255}
    eof
}

catch wait result
set exit_status [lindex $result 3]
exit $exit_status
